Execution Context Demonstration

Open the browser console (F12) to see the results of each demonstration.

1. Hoisting

Variable and function declarations are moved to the top of their scope.

// Variable Hoisting
console.log(x); // undefined (var is hoisted)
var x = 10;

// console.log(y); // ReferenceError (let is not hoisted)
let y = 20;

// const behaves like let
// console.log(z); // ReferenceError
const z = 30;

2. Function Hoisting

Function declarations are hoisted completely, expressions are not.

// Function Declaration - Hoisted
sayHello(); // Works! "Hello"

function sayHello() {
    console.log("Hello");
}

// Function Expression - NOT Hoisted
// sayGoodbye(); // TypeError: not a function

const sayGoodbye = function() {
    console.log("Goodbye");
};

// Arrow Function - NOT Hoisted
// greet(); // TypeError

const greet = () => console.log("Hi");

3. Call Stack

JavaScript uses a call stack to manage execution contexts.

function first() {
    console.log("First - Start");
    second();
    console.log("First - End");
}

function second() {
    console.log("Second - Start");
    third();
    console.log("Second - End");
}

function third() {
    console.log("Third - Only");
}

first();

// Output:
// First - Start
// Second - Start
// Third - Only
// Second - End
// First - End

4. Global vs Function Scope

Variables have different accessibility based on where they are declared.

// Global Scope
var globalVar = "I'm global (var)";
let globalLet = "I'm global (let)";
const globalConst = "I'm global (const)";

function checkScope() {
    // Function Scope
    var functionVar = "I'm function scoped";
    let functionLet = "I'm block scoped";
    
    console.log(globalVar); // Accessible
    console.log(functionVar); // Accessible
    
    if (true) {
        // Block Scope
        let blockLet = "I'm block scoped";
        var blockVar = "I'm function scoped (var ignores block)";
        console.log(blockLet); // Accessible here
    }
    
    // console.log(blockLet); // ReferenceError
    console.log(blockVar); // Accessible (var is function-scoped)
}

checkScope();
// console.log(functionVar); // ReferenceError

5. The 'this' Keyword

The value of 'this' depends on how a function is called.

// Global context
console.log(this); // Window object

// Object method
const person = {
    name: "Ali",
    greet: function() {
        console.log(this.name); // "Ali"
    },
    // Arrow function doesn't have its own 'this'
    greetArrow: () => {
        console.log(this.name); // undefined (window.name)
    }
};

person.greet();
person.greetArrow();

// Constructor function
function User(name) {
    this.name = name;
    this.display = function() {
        console.log(this.name);
    };
}

const user1 = new User("Mona");
user1.display(); // "Mona"

// Arrow in setTimeout (preserves 'this')
const obj = {
    name: "Object",
    delayed: function() {
        setTimeout(() => {
            console.log(this.name); // "Object"
        }, 100);
    }
};

6. ES6 Features Used in Project

Demonstration of ES6 features used throughout the project.

// 1. let & const
let mutableVar = "can change";
const immutableVar = "cannot reassign";

// 2. Arrow Functions
const add = (a, b) => a + b;
const square = x => x * x;

// 3. Template Literals
const name = "World";
console.log(`Hello, ${name}!`);

// 4. Destructuring
const [first, second] = [1, 2];
const { fname, age } = { fname: "Ali", age: 25 };

// 5. Spread Operator
const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5];
const obj1 = { a: 1 };
const obj2 = { ...obj1, b: 2 };

// 6. Default Parameters
const greet = (name = "Guest") => `Hello, ${name}`;

// 7. map, filter, reduce
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(n => n * 2);
const evens = numbers.filter(n => n % 2 === 0);
const sum = numbers.reduce((acc, n) => acc + n, 0);
